<script>
  // Try to open the deep link; if it fails, go to the web URL.
  function openApp(appUrl, webUrl){
    let opened = false;
    const t = Date.now();
    // Try by setting location to the app URL
    window.location.href = appUrl;
    // After 800ms, if still here, go to web fallback
    setTimeout(()=>{
      if (Date.now() - t < 1600) window.location.href = webUrl;
    }, 800);
  }
  document.querySelectorAll('.app').forEach(a=>{
    a.addEventListener('click', e=>{
      e.preventDefault();
      openApp(a.dataset.app, a.dataset.web);
    });
  });
</script>
